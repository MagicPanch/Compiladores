*{CODIGO DE TESTEO PARA EL ASSEMBLER DE SENTENCIAS FOR CON ANIDAMIENTO INCLUIDO}*

	INT i1; i2,
	ULONG u1; u2,

	i2 = 10_i,
	u2 = 3000_ul,

	FOR i1 IN RANGE (10_i; 0_i; -2_i) {		*{HAY 5 ITERACIONES DE ESTE FOR (YA QUE i1 SE INICIALIZA EN 10, SE LE RESTA DE A 2 Y ES HASTA QUE LLEGUE A 0)}*
		i2 = i2 + 1_i,				*{POR CADA UNA DE LAS 5 ITERACIONES DEL FOR MAS EXTERNO, SE EJECUTA ESTA INSTRUCCION, DADO QUE i2 SE INICIALIZA EN 10 Y HAY 5 ITERACIONES, SU VALOR FINAL DEBE SER 15}*
		FOR u1 IN RANGE (0_ul; 20_ul; 1_ul)	*{HAY 20 ITERACIONES DE ESTE FOR (YA QUE u1 SE INICIALIZA EN 0, SE LE SUMA DE A 1 Y ES HASTA QUE LLEGUE A 20). A SU VEZ ESTE FOR SE EJECUTA POR CADA UNA DE LAS 5 ITERACIONES DEL FOR MAS EXTERNO}*
			u2 -= 5_ul,			*{POR CADA UNA DE LAS 20 ITERACIONES DEL FOR INTERNO, PARA CADA UNA DE LAS 5 ITERACIONES DEL FOR EXTERNO, SE EJECUTA ESTA INSTRUCCION. EN TOTAL SON 100 ITERACIONES, DONDE EN CADA UNA SE LE RESTA 5 A LA VARIABLE u2 QUE ESTABA INICIALIZADA EN 3000. POR LO TANTO, SU VALOR FINAL DEBE SER 2500}*
	},
}

i2:main = 15 (POR LO EXPLICADO ANTES)
u1:main = 20 (YA QUE EN LAS 5 VECES QUE SE EJECUTA EL FOR MAS INTERNO, Y POR ENDE EN LA ULTIMA, AL FINALIZAR EL MISMO DICHA VARIABLE DE CONTROL VALE 20)
